1005  constexpr iterator insert(const_iterator position, const_reference value)
1006  {
1007  // The allocated node is in-place construced and recorded in every
1008  // following statements.
1009  internal_node_type *node = node_allocator.allocate(1);
1010 
1011  // Insert the new node...
1012  // ...before the position node...
1013  if (internal_node_type *position_node = position.node) {
1014  // ...as the new left child...
1015  if (position_node->parent) {
1016  std::construct_at(node, value, nullptr, nullptr,
1017  position_node->left_sibling, position_node,
1018  position_node->parent);
1019  position_node->left_sibling = node;
1020  // ...with a left sibling node.
1021  if (internal_node_type *left_node = node->left_sibling) {
1022  left_node->right_sibling = node;
1023  }
1024  // ...without a left sibling node.
1025  else {
1026  position_node->parent->first_child = node;
1027  }
1028  }
1029  // ...as the new root.
1030  else {
1031  std::construct_at(node, value, position_node, position_node);
1032  position_node->parent = node;
1033  root = node;
1034  }
1035  }
1036  // ...as the new last node...
1037  else {
1038  // ... as the last child of the root node.
1039  if (root) {
1040  std::construct_at(node, value, nullptr, nullptr, root->last_child,
1041  nullptr, root);
1042  if (!root->first_child) {
1043  root->first_child = node;
1044  }
1045  if (root->last_child) {
1046  root->last_child->right_sibling = node;
1047  }
1048  root->last_child = node;
1049  }
1050  // ...as the sole, and root node.
1051  else {
1052  std::construct_at(node, value);
1053  root = node;
1054  }
1055  }
1056 
1057  ++node_count;
1058  return { node };
1059  }

1468  constexpr void cleave(internal_node_type *node)
1469  {
1470  if (node == root) {

1473  if (node->parent) {

1477  if (node->parent->last_child == node) {
1478  node->parent->last_child = node->left_sibling;
1479  }
1480  }

1482  node->left_sibling->right_sibling = node->right_sibling;
1483  node->left_sibling = nullptr;
1484  }

1500  constexpr void prune(internal_node_type *node)
1501  {
1502  while (internal_node_type *current = node) {
1503  if (current->last_child) {
1504  if (current->right_sibling) {
1505  current->last_child->right_sibling = current->right_sibling;
1506  }
1507  node = current->first_child;
1508  } else if (current->right_sibling) {
1509  node = current->right_sibling;
1510  } else {
1511  node = nullptr;
1512  }
1513 
1514  std::destroy_at(current);
1515  node_allocator.deallocate(current, 1);
1516 
1517  --node_count;
1518  }
