1002  constexpr iterator insert(const_iterator position, const_reference value)
1003  {
1004  // The allocated node is in-place construced and recorded in every
1005  // following statements.
1006  internal_node_type *node = node_allocator.allocate(1);
1007 
1008  // Insert the new node...
1009  // ...before the position node...
1010  if (internal_node_type *position_node = position.node) {
1011  // ...as the new left child...
1012  if (position_node->parent) {
1013  std::construct_at(node, value, nullptr, nullptr,
1014  position_node->left_sibling, position_node,
1015  position_node->parent);
1016  position_node->left_sibling = node;
1017  // ...with a left sibling node.
1018  if (internal_node_type *left_node = node->left_sibling) {
1019  left_node->right_sibling = node;
1020  }
1021  // ...without a left sibling node.
1022  else {
1023  position_node->parent->first_child = node;
1024  }
1025  }
1026  // ...as the new root.
1027  else {
1028  std::construct_at(node, value, position_node, position_node);
1029  position_node->parent = node;
1030  root = node;
1031  }
1032  }
1033  // ...as the new last node...
1034  else {
1035  // ... as the last child of the root node.
1036  if (root) {
1037  std::construct_at(node, value, nullptr, nullptr, root->last_child,
1038  nullptr, root);
1039  if (!root->first_child) {
1040  root->first_child = node;
1041  }
1042  if (root->last_child) {
1043  root->last_child->right_sibling = node;
1044  }
1045  root->last_child = node;
1046  }
1047  // ...as the sole, and root node.
1048  else {
1049  std::construct_at(node, value);
1050  root = node;
1051  }
1052  }
1053 
1054  ++node_count;
1055 
1056  return { node };
1057  }

1505  constexpr void prune(internal_node_type *node)
1506  {
1507  while (internal_node_type *current = node) {
1508  if (current->last_child) {
1509  if (current->right_sibling) {
1510  current->last_child->right_sibling = current->right_sibling;
1511  }
1512  node = current->first_child;
1513  } else if (current->right_sibling) {
1514  node = current->right_sibling;
