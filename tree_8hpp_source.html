1470  internal_node_type *left_sibling,
1471  internal_node_type *parent)
1472  {

1474  if (other_node) {

1476  // allocator, the other node data value, and any left sibling or parent

1484  node->first_child = copy(other_node->first_child, nullptr, node);
1485  // ...reference the parent's last child with the last right sibling.
1486  if (node->parent && !node->parent->last_child) {
