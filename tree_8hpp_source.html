1011  constexpr iterator insert(const_iterator position, const_reference value)
1012  {
1013  // The allocated node is in-place construced and recorded in every
1014  // following statements.
1015  internal_node_type *node = node_allocator.allocate(1);
1016 
1017  // Insert the new node...
1018  // ...before the position node...
1019  if (internal_node_type *position_node = position.node) {
1020  // ...as the new left child...
1021  if (position_node->parent) {
1022  std::construct_at(node, value, nullptr, nullptr,
1023  position_node->left_sibling, position_node,
1024  position_node->parent);
1025  position_node->left_sibling = node;
1026  // ...with a left sibling node.
1027  if (internal_node_type *left_node = node->left_sibling) {
1028  left_node->right_sibling = node;
1029  }
1030  // ...without a left sibling node.
1031  else {
1032  position_node->parent->first_child = node;
1033  }
1034  }
1035  // ...as the new root.
1036  else {
1037  std::construct_at(node, value, position_node, position_node);
1038  position_node->parent = node;
1039  root = node;
1040  }
1041  }
1042  // ...as the new last node...
1043  else {
1044  // ... as the last child of the root node.
1045  if (root) {
1046  std::construct_at(node, value, nullptr, nullptr, root->last_child,
1047  nullptr, root);
1048  if (!root->first_child) {
1049  root->first_child = node;
1050  }
1051  if (root->last_child) {
1052  root->last_child->right_sibling = node;
1053  }
1054  root->last_child = node;
1055  }
1056  // ...as the sole, and root node.
1057  else {
1058  std::construct_at(node, value);
1059  root = node;
1060  }
1061  }
1062 
1063  ++node_count;
1064  return { node };
