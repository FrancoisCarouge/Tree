830  template <class... ArgumentsType>
831  constexpr iterator emplace(const_iterator position,
832  ArgumentsType &&... arguments)
833  {
834  // The allocated node is in-place construced and recorded in every
835  // following statements.
836  internal_node_type *node = node_allocator.allocate(1);
837 
838  // Insert the new node...
839  // ...before the position node...
840  if (internal_node_type *position_node = position.node) {
841  // ...as the new left child...
842  if (position_node->parent) {
843  std::construct_at(node, std::forward<ArgumentsType>(arguments)...,
844  nullptr, nullptr, position_node->left_sibling,
845  position_node, position_node->parent);
846  position_node->left_sibling = node;
847  // ...with a left sibling node.
848  if (internal_node_type *left_node = node->left_sibling) {
849  left_node->right_sibling = node;
850  }
851  // ...without a left sibling node.
852  else {
853  position_node->parent->first_child = node;
854  }
855  }
856  // ...as the new root.
857  else {
858  std::construct_at(node, std::forward<ArgumentsType>(arguments)...,
859  position_node, position_node);
860  position_node->parent = node;
861  root = node;
862  }
863  }
864  // ...as the new last node...
865  else {
866  // ... as the last child of the root node.
867  if (root) {
868  std::construct_at(node, std::forward<ArgumentsType>(arguments)...,
869  nullptr, nullptr, root->last_child, nullptr, root);
870  if (!root->first_child) {
871  root->first_child = node;
872  }
873  if (root->last_child) {
874  root->last_child->right_sibling = node;
875  }
876  root->last_child = node;
877  }
878  // ...as the sole, and root node.
879  else {
880  std::construct_at(node, std::forward<ArgumentsType>(arguments)...);
881  root = node;
882  }
883  }
