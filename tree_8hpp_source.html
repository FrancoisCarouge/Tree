1039  constexpr iterator insert(const_iterator position, const_reference value)
1040  {
1041  // The allocated node is in-place construced and recorded in every
1042  // following statements.
1043  internal_node_type *node = node_allocator.allocate(1);
1044 
1045  // Insert the new node...
1046  // ...before the position node...
1047  if (internal_node_type *position_node = position.node) {
1048  // ...as the new left child...
1049  if (position_node->parent) {
1050  std::construct_at(node, value, nullptr, nullptr,
1051  position_node->left_sibling, position_node,
1052  position_node->parent);
1053  position_node->left_sibling = node;
1054  // ...with a left sibling node.
1055  if (internal_node_type *left_node = node->left_sibling) {
1056  left_node->right_sibling = node;
1057  }
1058  // ...without a left sibling node.
1059  else {
1060  position_node->parent->first_child = node;
1061  }
1062  }
1063  // ...as the new root.
1064  else {
1065  std::construct_at(node, value, position_node, position_node);
1066  position_node->parent = node;
1067  root = node;
1068  }
1069  }
1070  // ...as the new last node...
1071  else {
1072  // ... as the last child of the root node.
1073  if (root) {
1074  std::construct_at(node, value, nullptr, nullptr, root->last_child,
1075  nullptr, root);
1076  if (!root->first_child) {
1077  root->first_child = node;
1078  }
1079  if (root->last_child) {
1080  root->last_child->right_sibling = node;
1081  }
1082  root->last_child = node;
1083  }
1084  // ...as the sole, and root node.
1085  else {
1086  std::construct_at(node, value);
1087  root = node;
1088  }
1089  }
1090 
1091  ++node_count;
1092  return { node };
1093  }
