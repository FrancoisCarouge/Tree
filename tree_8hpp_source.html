1084  constexpr iterator insert(const_iterator position, const_reference value)
1085  {
1086  // The allocated node is in-place construced and recorded in every
1087  // following statements.
1088  internal_node_type *node = node_allocator.allocate(1);
1089 
1090  // Insert the new node...
1091  // ...before the position node...
1092  if (internal_node_type *position_node = position.node) {
1093  // ...as the new left child...
1094  if (position_node->parent) {
1095  std::construct_at(node, value, nullptr, nullptr,
1096  position_node->left_sibling, position_node,
1097  position_node->parent);
1098  position_node->left_sibling = node;
1099  // ...with a left sibling node.
1100  if (internal_node_type *left_node = node->left_sibling) {
1101  left_node->right_sibling = node;
1102  }
1103  // ...without a left sibling node.
1104  else {
1105  position_node->parent->first_child = node;
1106  }
1107  }
1108  // ...as the new root.
1109  else {
1110  std::construct_at(node, value, position_node, position_node);
1111  position_node->parent = node;
1112  root = node;
1113  }
1114  }
1115  // ...as the new last node...
1116  else {
1117  // ... as the last child of the root node.
1118  if (root) {
1119  std::construct_at(node, value, nullptr, nullptr, root->last_child,
1120  nullptr, root);
1121  if (!root->first_child) {
1122  root->first_child = node;
1123  }
1124  if (root->last_child) {
1125  root->last_child->right_sibling = node;
1126  }
1127  root->last_child = node;
1128  }
1129  // ...as the sole, and root node.
1130  else {
1131  std::construct_at(node, value);
1132  root = node;
1133  }
1134  }
1135 
1136  ++node_count;
1137 
1138  return { node };
1139  }
