774  template <class... ArgumentsType>
775  constexpr iterator emplace(const_iterator position,
776  ArgumentsType &&... arguments)
777  {
778  // The allocated node is in-place construced and recorded in every
779  // following statements.
780  internal_node_type *node = node_allocator.allocate(1);
781 
782  // Insert the new node...
783  // ...before the position node...
784  if (internal_node_type *position_node = position.node) {
785  // ...as the new left child...
786  if (position_node->parent) {
787  std::construct_at(node, std::forward<ArgumentsType>(arguments)...,
788  nullptr, nullptr, position_node->left_sibling,
789  position_node, position_node->parent);
790  position_node->left_sibling = node;
791  // ...with a left sibling node.
792  if (internal_node_type *left_node = node->left_sibling) {
793  left_node->right_sibling = node;
794  }
795  // ...without a left sibling node.
796  else {
797  position_node->parent->first_child = node;
798  }
799  }
800  // ...as the new root.
801  else {
802  std::construct_at(node, std::forward<ArgumentsType>(arguments)...,
803  position_node, position_node);
804  position_node->parent = node;
805  root = node;
806  }
807  }
808  // ...as the new last node...
809  else {
810  // ... as the sole, first, and last child of the previous last node.
811  if (last) {
812  std::construct_at(node, std::forward<ArgumentsType>(arguments)...,
813  nullptr, nullptr, nullptr, nullptr, last);
814  last->first_child = node;
815  last->last_child = node;
816  last = node;
817  }
818  // ...as the sole, last, and root node.
819  else {
820  std::construct_at(node, std::forward<ArgumentsType>(arguments)...);
821  root = node;
822  last = node;
823  }
824  }
825 
826  ++node_count;
827  return iterator{ node };
828  }

844  constexpr iterator erase(iterator position)
845  {
846  // Identify the node following this subtree which will become the next node.
847  iterator next{ position.node->next_ancestor_sibling() };
848 
849  // Orphan the node.
850  if (position.node->parent) {
851  if (position.node->parent->first_child == position.node) {
852  position.node->parent->first_child = position.node->right_sibling;
853  }
854  if (position.node->parent->last_child == position.node) {
855  position.node->parent->last_child = position.node->left_sibling;
856  }
857  }
858 
859  // Isolate the node from its siblings.
860  if (position.node->left_sibling) {
861  position.node->left_sibling->right_sibling = position.node->right_sibling;
862  position.node->left_sibling = nullptr;
863  }
864  if (position.node->right_sibling) {
865  position.node->right_sibling->left_sibling = position.node->left_sibling;
866  position.node->right_sibling = nullptr;
867  }
868 
869  if (position.node == root) {
870  root = nullptr;
871  }
872 
873  if (position.node == last) {
874  last = nullptr;
875  }
876 
877  erase(position.node);
878 
879  return next;
880  }
