900  template <class... ArgumentsType>
901  constexpr iterator emplace(const_iterator position,
902  ArgumentsType &&... arguments)
903  {
904  // The allocated node is in-place construced and recorded in every
905  // following statements.
906  internal_node_type *node = node_allocator.allocate(1);
907 
908  // Insert the new node...
909  // ...before the position node...
910  if (internal_node_type *position_node = position.node) {
911  // ...as the new left child...
912  if (position_node->parent) {
913  std::construct_at(node, std::forward<ArgumentsType>(arguments)...,
914  nullptr, nullptr, position_node->left_sibling,
915  position_node, position_node->parent);
916  position_node->left_sibling = node;
917  // ...with a left sibling node.
918  if (internal_node_type *left_node = node->left_sibling) {
919  left_node->right_sibling = node;
920  }
921  // ...without a left sibling node.
922  else {
923  position_node->parent->first_child = node;
924  }
925  }
926  // ...as the new root.
927  else {
928  std::construct_at(node, std::forward<ArgumentsType>(arguments)...,
929  position_node, position_node);
930  position_node->parent = node;
931  root = node;
932  }
933  }
934  // ...as the new last node...
935  else {
936  // ... as the sole, first, and last child of the previous last node.
937  if (last) {
938  std::construct_at(node, std::forward<ArgumentsType>(arguments)...,
939  nullptr, nullptr, nullptr, nullptr, last);
940  last->first_child = node;
941  last->last_child = node;
942  last = node;
943  }
944  // ...as the sole, last, and root node.
945  else {
946  std::construct_at(node, std::forward<ArgumentsType>(arguments)...);
947  root = node;
948  last = node;
949  }
950  }
951 
952  ++node_count;
953  return iterator{ node };
954  }

970  constexpr iterator erase(iterator position)
971  {
972  // Identify the node following this subtree which will become the next node.
973  iterator next{ position.node->next_ancestor_sibling() };
974 
975  // Orphan the node.
976  if (position.node->parent) {
977  if (position.node->parent->first_child == position.node) {
978  position.node->parent->first_child = position.node->right_sibling;
979  }
980  if (position.node->parent->last_child == position.node) {
981  position.node->parent->last_child = position.node->left_sibling;
982  }
983  }
984 
985  // Isolate the node from its siblings.
986  if (position.node->left_sibling) {
987  position.node->left_sibling->right_sibling = position.node->right_sibling;
988  position.node->left_sibling = nullptr;
989  }
990  if (position.node->right_sibling) {
991  position.node->right_sibling->left_sibling = position.node->left_sibling;
992  position.node->right_sibling = nullptr;
993  }
994 
995  if (position.node == root) {
996  root = nullptr;
997  }
998 
999  if (position.node == last) {
1000  last = nullptr;
1001  }
1002 
1003  erase(position.node);
1004 
1005  return next;
1006  }
