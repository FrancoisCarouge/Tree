817  constexpr iterator emplace(const_iterator position,
818  ArgumentsType &&... arguments)
819  {
820  // The allocated node is in-place construced and recorded in every
821  // following statements.
822  internal_node_type *node = node_allocator.allocate(1);
823 
824  // Insert the new node...
825  // ...before the position node...
826  if (internal_node_type *position_node = position.node) {
827  // ...as the new left child...
828  if (position_node->parent) {
829  std::construct_at(node, std::forward<ArgumentsType>(arguments)...,
830  nullptr, nullptr, position_node->left_sibling,
831  position_node, position_node->parent);
832  position_node->left_sibling = node;
833  // ...with a left sibling node.
834  if (internal_node_type *left_node = node->left_sibling) {
835  left_node->right_sibling = node;
836  }
837  // ...without a left sibling node.
838  else {
839  position_node->parent->first_child = node;
840  }
841  }
842  // ...as the new root.
843  else {
844  std::construct_at(node, std::forward<ArgumentsType>(arguments)...,
845  position_node, position_node);
846  position_node->parent = node;
847  root = node;
848  }
849  }

851  else {
