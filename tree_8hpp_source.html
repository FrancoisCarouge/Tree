1061  constexpr iterator insert(const_iterator position, const_reference value)
1062  {
1063  // The allocated node is in-place construced and recorded in every
1064  // following statements.
1065  internal_node_type *node = node_allocator.allocate(1);
1066 
1067  // Insert the new node...
1068  // ...before the position node...
1069  if (internal_node_type *position_node = position.node) {
1070  // ...as the new left child...
1071  if (position_node->parent) {
1072  std::construct_at(node, value, nullptr, nullptr,
1073  position_node->left_sibling, position_node,
1074  position_node->parent);
1075  position_node->left_sibling = node;
1076  // ...with a left sibling node.
1077  if (internal_node_type *left_node = node->left_sibling) {
1078  left_node->right_sibling = node;
1079  }
1080  // ...without a left sibling node.
1081  else {
1082  position_node->parent->first_child = node;
1083  }
1084  }
1085  // ...as the new root.
1086  else {
1087  std::construct_at(node, value, position_node, position_node);
1088  position_node->parent = node;
1089  root = node;
1090  }
1091  }
1092  // ...as the new last node...
1093  else {
1094  // ... as the last child of the root node.
1095  if (root) {
1096  std::construct_at(node, value, nullptr, nullptr, root->last_child,
1097  nullptr, root);
1098  if (!root->first_child) {
1099  root->first_child = node;
1100  }
1101  if (root->last_child) {
1102  root->last_child->right_sibling = node;
1103  }
1104  root->last_child = node;
1105  }
1106  // ...as the sole, and root node.
1107  else {
1108  std::construct_at(node, value);
1109  root = node;
1110  }
1111  }
1112 
1113  ++node_count;
1114  return { node };
1115  }
