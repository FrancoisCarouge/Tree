1083  constexpr iterator insert(const_iterator position, const_reference value)
1084  {
1085  // The allocated node is in-place construced and recorded in every
1086  // following statements.
1087  internal_node_type *node = node_allocator.allocate(1);
1088 
1089  // Insert the new node...
1090  // ...before the position node...
1091  if (internal_node_type *position_node = position.node) {
1092  // ...as the new left child...
1093  if (position_node->parent) {
1094  std::construct_at(node, value, nullptr, nullptr,
1095  position_node->left_sibling, position_node,
1096  position_node->parent);
1097  position_node->left_sibling = node;
1098  // ...with a left sibling node.
1099  if (internal_node_type *left_node = node->left_sibling) {
1100  left_node->right_sibling = node;
1101  }
1102  // ...without a left sibling node.
1103  else {
1104  position_node->parent->first_child = node;
1105  }
1106  }
1107  // ...as the new root.
1108  else {
1109  std::construct_at(node, value, position_node, position_node);
1110  position_node->parent = node;
1111  root = node;
1112  }
1113  }
1114  // ...as the new last node...
1115  else {
1116  // ... as the last child of the root node.
1117  if (root) {
1118  std::construct_at(node, value, nullptr, nullptr, root->last_child,
1119  nullptr, root);
1120  if (!root->first_child) {
1121  root->first_child = node;
1122  }
1123  if (root->last_child) {
1124  root->last_child->right_sibling = node;
1125  }
1126  root->last_child = node;
1127  }
1128  // ...as the sole, and root node.
1129  else {
1130  std::construct_at(node, value);
1131  root = node;
1132  }
1133  }
1134 
1135  ++node_count;
1136 
1137  return { node };
1138  }
