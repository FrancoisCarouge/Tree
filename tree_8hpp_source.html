971  {

973  // following statements.
974  internal_node_type *node = node_allocator.allocate(1);
975

978  if (internal_node_type *position_node = position.node) {

982  nullptr, nullptr, position_node->left_sibling,

985  // ...with a left sibling node.
986  if (internal_node_type *left_node = node->left_sibling) {

993  }

996  std::construct_at(node, std::forward<ArgumentsType>(arguments)...,
997  position_node, position_node);
998  position_node->parent = node;
