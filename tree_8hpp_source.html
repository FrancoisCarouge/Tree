881  template <class... ArgumentsType>
882  constexpr iterator emplace(const_iterator position,
883  ArgumentsType &&... arguments)
884  {
885  // The allocated node is in-place construced and recorded in every
886  // following statements.
887  internal_node_type *node = node_allocator.allocate(1);
888 
889  // Insert the new node...
890  // ...before the position node...
891  if (internal_node_type *position_node = position.node) {
892  // ...as the new left child...
893  if (position_node->parent) {
894  std::construct_at(node, std::forward<ArgumentsType>(arguments)...,
895  nullptr, nullptr, position_node->left_sibling,
896  position_node, position_node->parent);
897  position_node->left_sibling = node;
898  // ...with a left sibling node.
899  if (internal_node_type *left_node = node->left_sibling) {
900  left_node->right_sibling = node;
901  }
902  // ...without a left sibling node.
903  else {
904  position_node->parent->first_child = node;
905  }
906  }
907  // ...as the new root.
908  else {
909  std::construct_at(node, std::forward<ArgumentsType>(arguments)...,
910  position_node, position_node);
911  position_node->parent = node;
912  root = node;
913  }
914  }
915  // ...as the new last node...
916  else {
917  // ... as the sole, first, and last child of the previous last node.
918  if (last) {
919  std::construct_at(node, std::forward<ArgumentsType>(arguments)...,
920  nullptr, nullptr, nullptr, nullptr, last);
921  last->first_child = node;
922  last->last_child = node;
923  last = node;
924  }
925  // ...as the sole, last, and root node.
926  else {
927  std::construct_at(node, std::forward<ArgumentsType>(arguments)...);
928  root = node;
929  last = node;
930  }
931  }
932 
933  ++node_count;
934  return iterator{ node };
935  }

951  constexpr iterator erase(iterator position)
952  {
953  // Identify the node following this subtree which will become the next node.
954  iterator next{ position.node->next_ancestor_sibling() };
955 
956  // Orphan the node.
957  if (position.node->parent) {
958  if (position.node->parent->first_child == position.node) {
959  position.node->parent->first_child = position.node->right_sibling;
960  }
961  if (position.node->parent->last_child == position.node) {
962  position.node->parent->last_child = position.node->left_sibling;
963  }
964  }
965 
966  // Isolate the node from its siblings.
967  if (position.node->left_sibling) {
968  position.node->left_sibling->right_sibling = position.node->right_sibling;
969  position.node->left_sibling = nullptr;
970  }
971  if (position.node->right_sibling) {
972  position.node->right_sibling->left_sibling = position.node->left_sibling;
973  position.node->right_sibling = nullptr;
974  }
975 
976  if (position.node == root) {
977  root = nullptr;
978  }
979 
980  if (position.node == last) {
981  last = nullptr;
982  }
983 
984  erase(position.node);
985 
986  return next;
987  }
