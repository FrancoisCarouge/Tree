1013  constexpr iterator insert(const_iterator position, const_reference value)
1014  {
1015  // The allocated node is in-place construced and recorded in every
1016  // following statements.
1017  internal_node_type *node = node_allocator.allocate(1);
1018 
1019  // Insert the new node...
1020  // ...before the position node...
1021  if (internal_node_type *position_node = position.node) {
1022  // ...as the new left child...
1023  if (position_node->parent) {
1024  std::construct_at(node, value, nullptr, nullptr,
1025  position_node->left_sibling, position_node,
1026  position_node->parent);
1027  position_node->left_sibling = node;
1028  // ...with a left sibling node.
1029  if (internal_node_type *left_node = node->left_sibling) {
1030  left_node->right_sibling = node;
1031  }
1032  // ...without a left sibling node.
1033  else {
1034  position_node->parent->first_child = node;
1035  }
1036  }
1037  // ...as the new root.
1038  else {
1039  std::construct_at(node, value, position_node, position_node);
1040  position_node->parent = node;
1041  root = node;
1042  }
1043  }
1044  // ...as the new last node...
1045  else {
1046  // ... as the last child of the root node.
1047  if (root) {
1048  std::construct_at(node, value, nullptr, nullptr, root->last_child,
1049  nullptr, root);
1050  if (!root->first_child) {
1051  root->first_child = node;
1052  }
1053  if (root->last_child) {
1054  root->last_child->right_sibling = node;
1055  }
1056  root->last_child = node;
1057  }
1058  // ...as the sole, and root node.
1059  else {
1060  std::construct_at(node, value);
1061  root = node;
1062  }
1063  }
1064 
1065  ++node_count;
1066  return { node };
1067  }

1481  {
