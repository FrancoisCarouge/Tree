918  template <class... ArgumentsType>
919  constexpr iterator emplace(const_iterator position,
920  ArgumentsType &&... arguments)
921  {
922  // The allocated node is in-place construced and recorded in every
923  // following statements.
924  internal_node_type *node = node_allocator.allocate(1);
925 
926  // Insert the new node...
927  // ...before the position node...
928  if (internal_node_type *position_node = position.node) {
929  // ...as the new left child...
930  if (position_node->parent) {
931  std::construct_at(node, std::forward<ArgumentsType>(arguments)...,
932  nullptr, nullptr, position_node->left_sibling,
933  position_node, position_node->parent);
934  position_node->left_sibling = node;
935  // ...with a left sibling node.
936  if (internal_node_type *left_node = node->left_sibling) {
937  left_node->right_sibling = node;
938  }
939  // ...without a left sibling node.
940  else {
941  position_node->parent->first_child = node;
942  }
943  }
944  // ...as the new root.
945  else {
946  std::construct_at(node, std::forward<ArgumentsType>(arguments)...,
947  position_node, position_node);
948  position_node->parent = node;
949  root = node;
950  }
951  }
952  // ...as the new last node...
953  else {
954  // ... as the sole, first, and last child of the previous last node.
955  if (last) {
956  std::construct_at(node, std::forward<ArgumentsType>(arguments)...,
957  nullptr, nullptr, nullptr, nullptr, last);
958  last->first_child = node;
959  last->last_child = node;
960  last = node;
961  }
962  // ...as the sole, last, and root node.
963  else {
964  std::construct_at(node, std::forward<ArgumentsType>(arguments)...);
965  root = node;
966  last = node;
967  }
968  }
969 
970  ++node_count;
971  return iterator{ node };
972  }

988  constexpr iterator erase(iterator position)
989  {
990  // Identify the node following this subtree which will become the next node.
991  iterator next{ position.node->next_ancestor_sibling() };
992 
993  // Orphan the node.
994  if (position.node->parent) {
995  if (position.node->parent->first_child == position.node) {
996  position.node->parent->first_child = position.node->right_sibling;
997  }
998  if (position.node->parent->last_child == position.node) {
999  position.node->parent->last_child = position.node->left_sibling;
1000  }
1001  }
1002 
1003  // Isolate the node from its siblings.
1004  if (position.node->left_sibling) {
1005  position.node->left_sibling->right_sibling = position.node->right_sibling;
1006  position.node->left_sibling = nullptr;
1007  }
1008  if (position.node->right_sibling) {
1009  position.node->right_sibling->left_sibling = position.node->left_sibling;
1010  position.node->right_sibling = nullptr;
1011  }
1012 
1013  if (position.node == root) {
1014  root = nullptr;
1015  }
1016 
1017  if (position.node == last) {
1018  last = nullptr;
1019  }
1020 
1021  erase(position.node);
1022 
1023  return next;
1024  }
