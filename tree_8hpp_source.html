1017  constexpr iterator insert(const_iterator position, const_reference value)
1018  {
1019  // The allocated node is in-place construced and recorded in every
1020  // following statements.
1021  internal_node_type *node = node_allocator.allocate(1);
1022 
1023  // Insert the new node...
1024  // ...before the position node...
1025  if (internal_node_type *position_node = position.node) {
1026  // ...as the new left child...
1027  if (position_node->parent) {
1028  std::construct_at(node, value, nullptr, nullptr,
1029  position_node->left_sibling, position_node,
1030  position_node->parent);
1031  position_node->left_sibling = node;
1032  // ...with a left sibling node.
1033  if (internal_node_type *left_node = node->left_sibling) {
1034  left_node->right_sibling = node;
1035  }
1036  // ...without a left sibling node.
1037  else {
1038  position_node->parent->first_child = node;
1039  }
1040  }
1041  // ...as the new root.
1042  else {
1043  std::construct_at(node, value, position_node, position_node);
1044  position_node->parent = node;
1045  root = node;
1046  }
1047  }
1048  // ...as the new last node...
1049  else {
1050  // ... as the last child of the root node.
1051  if (root) {
1052  std::construct_at(node, value, nullptr, nullptr, root->last_child,
1053  nullptr, root);
1054  if (!root->first_child) {
1055  root->first_child = node;
1056  }
1057  if (root->last_child) {
1058  root->last_child->right_sibling = node;
1059  }
1060  root->last_child = node;
1061  }
1062  // ...as the sole, and root node.
1063  else {
1064  std::construct_at(node, value);
1065  root = node;
1066  }
1067  }
1068 
1069  ++node_count;
1070 
1071  return { node };
1072  }

1489  {

1520  constexpr void prune(internal_node_type *node)
1521  {
1522  while (internal_node_type *current = node) {
1523  if (current->last_child) {
1524  if (current->right_sibling) {
1525  current->last_child->right_sibling = current->right_sibling;
1526  }
1527  node = current->first_child;
1528  } else if (current->right_sibling) {
1529  node = current->right_sibling;
1530  } else {
1531  node = nullptr;
1532  }
1533

1535  node_allocator.deallocate(current, 1);
1536 
1537  --node_count;
