1048  constexpr iterator insert(const_iterator position, const_reference value)
1049  {
1050  // The allocated node is in-place construced and recorded in every
1051  // following statements.
1052  internal_node_type *node = node_allocator.allocate(1);
1053 
1054  // Insert the new node...
1055  // ...before the position node...
1056  if (internal_node_type *position_node = position.node) {
1057  // ...as the new left child...
1058  if (position_node->parent) {
1059  std::construct_at(node, value, nullptr, nullptr,
1060  position_node->left_sibling, position_node,
1061  position_node->parent);
1062  position_node->left_sibling = node;
1063  // ...with a left sibling node.
1064  if (internal_node_type *left_node = node->left_sibling) {
1065  left_node->right_sibling = node;
1066  }
1067  // ...without a left sibling node.
1068  else {
1069  position_node->parent->first_child = node;
1070  }
1071  }
1072  // ...as the new root.
1073  else {
1074  std::construct_at(node, value, position_node, position_node);
1075  position_node->parent = node;
1076  root = node;
1077  }
1078  }
1079  // ...as the new last node...
1080  else {
1081  // ... as the last child of the root node.
1082  if (root) {
1083  std::construct_at(node, value, nullptr, nullptr, root->last_child,
1084  nullptr, root);
1085  if (!root->first_child) {
1086  root->first_child = node;
1087  }
1088  if (root->last_child) {
1089  root->last_child->right_sibling = node;
1090  }
1091  root->last_child = node;
1092  }
1093  // ...as the sole, and root node.
1094  else {
1095  std::construct_at(node, value);
1096  root = node;
1097  }
1098  }
1099 
1100  ++node_count;
1101  return { node };
1102  }
