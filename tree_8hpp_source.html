1075  constexpr iterator insert(const_iterator position, const_reference value)
1076  {
1077  // The allocated node is in-place construced and recorded in every
1078  // following statements.
1079  internal_node_type *node = node_allocator.allocate(1);
1080 
1081  // Insert the new node...
1082  // ...before the position node...
1083  if (internal_node_type *position_node = position.node) {
1084  // ...as the new left child...
1085  if (position_node->parent) {
1086  std::construct_at(node, value, nullptr, nullptr,
1087  position_node->left_sibling, position_node,
1088  position_node->parent);
1089  position_node->left_sibling = node;
1090  // ...with a left sibling node.
1091  if (internal_node_type *left_node = node->left_sibling) {
1092  left_node->right_sibling = node;
1093  }
1094  // ...without a left sibling node.
1095  else {
1096  position_node->parent->first_child = node;
1097  }
1098  }
1099  // ...as the new root.
1100  else {
1101  std::construct_at(node, value, position_node, position_node);
1102  position_node->parent = node;
1103  root = node;
1104  }
1105  }
1106  // ...as the new last node...
1107  else {
1108  // ... as the last child of the root node.
1109  if (root) {
1110  std::construct_at(node, value, nullptr, nullptr, root->last_child,
1111  nullptr, root);
1112  if (!root->first_child) {
1113  root->first_child = node;
1114  }
1115  if (root->last_child) {
1116  root->last_child->right_sibling = node;
1117  }
1118  root->last_child = node;
1119  }
1120  // ...as the sole, and root node.
1121  else {
1122  std::construct_at(node, value);
1123  root = node;
1124  }
1125  }
1126

1128
