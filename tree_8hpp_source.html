1051  constexpr iterator insert(const_iterator position, const_reference value)
1052  {
1053  // The allocated node is in-place construced and recorded in every
1054  // following statements.
1055  internal_node_type *node = node_allocator.allocate(1);
1056 
1057  // Insert the new node...
1058  // ...before the position node...
1059  if (internal_node_type *position_node = position.node) {
1060  // ...as the new left child...
1061  if (position_node->parent) {
1062  std::construct_at(node, value, nullptr, nullptr,
1063  position_node->left_sibling, position_node,
1064  position_node->parent);
1065  position_node->left_sibling = node;
1066  // ...with a left sibling node.
1067  if (internal_node_type *left_node = node->left_sibling) {
1068  left_node->right_sibling = node;
1069  }
1070  // ...without a left sibling node.
1071  else {
1072  position_node->parent->first_child = node;
1073  }
1074  }
1075  // ...as the new root.
1076  else {
1077  std::construct_at(node, value, position_node, position_node);
1078  position_node->parent = node;
1079  root = node;
1080  }
1081  }
1082  // ...as the new last node...
1083  else {
1084  // ... as the last child of the root node.
1085  if (root) {
1086  std::construct_at(node, value, nullptr, nullptr, root->last_child,
1087  nullptr, root);
1088  if (!root->first_child) {
1089  root->first_child = node;
1090  }
1091  if (root->last_child) {
1092  root->last_child->right_sibling = node;
1093  }
1094  root->last_child = node;
1095  }
1096  // ...as the sole, and root node.
1097  else {
1098  std::construct_at(node, value);
1099  root = node;
1100  }
1101  }
1102 
1103  ++node_count;
1104 
1105  return { node };
1106  }

1555  constexpr void prune(internal_node_type *node)
1556  {
1557  while (internal_node_type *current = node) {
1558  if (current->last_child) {
1559  if (current->right_sibling) {
1560  current->last_child->right_sibling = current->right_sibling;
1561  }
1562  node = current->first_child;
