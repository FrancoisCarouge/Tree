878  template <class... ArgumentsType>
879  constexpr iterator emplace(const_iterator position,
880  ArgumentsType &&... arguments)
881  {
882  // The allocated node is in-place construced and recorded in every
883  // following statements.
884  internal_node_type *node = node_allocator.allocate(1);
885 
886  // Insert the new node...
887  // ...before the position node...
888  if (internal_node_type *position_node = position.node) {
889  // ...as the new left child...
890  if (position_node->parent) {
891  std::construct_at(node, std::forward<ArgumentsType>(arguments)...,
892  nullptr, nullptr, position_node->left_sibling,
893  position_node, position_node->parent);
894  position_node->left_sibling = node;
895  // ...with a left sibling node.
896  if (internal_node_type *left_node = node->left_sibling) {
897  left_node->right_sibling = node;
898  }
899  // ...without a left sibling node.
900  else {
901  position_node->parent->first_child = node;
902  }
903  }
904  // ...as the new root.
905  else {
906  std::construct_at(node, std::forward<ArgumentsType>(arguments)...,
907  position_node, position_node);
908  position_node->parent = node;
909  root = node;
910  }
911  }
912  // ...as the new last node...
913  else {
914  // ... as the sole, first, and last child of the previous last node.
915  if (last) {
916  std::construct_at(node, std::forward<ArgumentsType>(arguments)...,
917  nullptr, nullptr, nullptr, nullptr, last);
918  last->first_child = node;
919  last->last_child = node;
920  last = node;
921  }
922  // ...as the sole, last, and root node.
923  else {
924  std::construct_at(node, std::forward<ArgumentsType>(arguments)...);
925  root = node;
926  last = node;
927  }
928  }
929 
930  node_count++;
931  return iterator{ node };
932  }

948  constexpr iterator erase(iterator position)
949  {
950  // Identify the node following this subtree which will become the next node.
951  iterator next{ position.node->next_ancestor_sibling() };
952 
953  // Orphan the node.
954  if (position.node->parent) {
955  if (position.node->parent->first_child == position.node) {
956  position.node->parent->first_child = position.node->right_sibling;
957  }
958  if (position.node->parent->last_child == position.node) {
959  position.node->parent->last_child = position.node->left_sibling;
960  }
961  }
962 
963  // Isolate the node from its siblings.
964  if (position.node->left_sibling) {
965  position.node->left_sibling->right_sibling = position.node->right_sibling;
966  position.node->left_sibling = nullptr;
967  }
968  if (position.node->right_sibling) {
969  position.node->right_sibling->left_sibling = position.node->left_sibling;
970  position.node->right_sibling = nullptr;
971  }
972 
973  if (position.node == root) {
974  root = nullptr;
975  }
976 
977  if (position.node == last) {
978  last = nullptr;
979  }
980 
981  erase(position.node);
982 
983  return next;
984  }
