954  constexpr reference emplace_front(ArgumentsType &&... arguments)
955  {

957  // following statements.

959

997  template <class... ArgumentsType>
998  constexpr iterator emplace(const_iterator position,
999  ArgumentsType &&... arguments)
1000  {
1001  // The allocated node is in-place construced and recorded in every
1002  // following statements.
1003  internal_node_type *node = node_allocator.allocate(1);
1004 
1005  // Insert the new node...
1006  // ...before the position node...
1007  if (internal_node_type *position_node = position.node) {
1008  // ...as the new left child...
1009  if (position_node->parent) {
1010  std::construct_at(node, std::forward<ArgumentsType>(arguments)...,
1011  nullptr, nullptr, position_node->left_sibling,
1012  position_node, position_node->parent);
1013  position_node->left_sibling = node;
1014  // ...with a left sibling node.
1015  if (internal_node_type *left_node = node->left_sibling) {
1016  left_node->right_sibling = node;
1017  }
1018  // ...without a left sibling node.
1019  else {
1020  position_node->parent->first_child = node;
1021  }
1022  }
1023  // ...as the new root.
1024  else {
1025  std::construct_at(node, std::forward<ArgumentsType>(arguments)...,
1026  position_node, position_node);
1027  position_node->parent = node;
1028  root = node;
1029  }
1030  }
1031  // ...as the new last node...
1032  else {
1033  // ... as the last child of the root node.

1035  std::construct_at(node, std::forward<ArgumentsType>(arguments)...,
1036  nullptr, nullptr, root->last_child, nullptr, root);

1039  }
1040  if (root->last_child) {

1466  internal_node_type *copy(internal_node_type *other_node,
1467  internal_node_type *left_sibling,
1468  internal_node_type *parent)
1469  {
1470  // If there is another node to copy...
1471  if (other_node) {
1472  // ...allocate and in-place construct a node with this container's
1473  // allocator, the other node data value, and any left sibling or parent
1474  // pointers...
1475  internal_node_type *node = node_allocator.allocate(1);
1476  std::construct_at(node, other_node->data, nullptr, nullptr, left_sibling,
1477  nullptr, parent);
1478  // ...recursively copy the right sibling and first child passing in left
1479  // sibling and parent pointers...
1480  node->right_sibling = copy(other_node->right_sibling, node, node->parent);
1481  node->first_child = copy(other_node->first_child, nullptr, node);
1482  // ...reference the parent's last child with the last right sibling.
1483  if (node->parent && !node->parent->last_child) {
1484  node->parent->last_child = node;
1485  }
1486  return node;
1487  }
1488
