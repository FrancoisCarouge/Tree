817  template <class... ArgumentsType>
818  constexpr iterator emplace(const_iterator position,
819  ArgumentsType &&... arguments)
820  {
821  // The allocated node is in-place construced and recorded in every
822  // following statements.
823  internal_node_type *node = node_allocator.allocate(1);
824 
825  // Insert the new node...
826  // ...before the position node...
827  if (internal_node_type *position_node = position.node) {
828  // ...as the new left child...
829  if (position_node->parent) {
830  std::construct_at(node, std::forward<ArgumentsType>(arguments)...,
831  nullptr, nullptr, position_node->left_sibling,
832  position_node, position_node->parent);
833  position_node->left_sibling = node;
834  // ...with a left sibling node.
835  if (internal_node_type *left_node = node->left_sibling) {
836  left_node->right_sibling = node;
837  }
838  // ...without a left sibling node.
839  else {
840  position_node->parent->first_child = node;
841  }
842  }
843  // ...as the new root.
844  else {
845  std::construct_at(node, std::forward<ArgumentsType>(arguments)...,
846  position_node, position_node);
847  position_node->parent = node;
848  root = node;
849  }
850  }
851  // ...as the new last node...
852  else {
853  // ... as the sole, first, and last child of the previous last node.
854  if (last) {
855  std::construct_at(node, std::forward<ArgumentsType>(arguments)...,
856  nullptr, nullptr, nullptr, nullptr, last);
857  last->first_child = node;
858  last->last_child = node;
859  last = node;
860  }
861  // ...as the sole, last, and root node.
862  else {
863  std::construct_at(node, std::forward<ArgumentsType>(arguments)...);
864  root = node;
865  last = node;
866  }
867  }

892  iterator next{ position.node->next_ancestor_sibling() };
893
